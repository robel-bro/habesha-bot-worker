"use client";

import { motion } from "framer-motion";
import { useInView } from "react-intersection-observer";
import Image from "next/image";
import Link from "next/link";

function Reveal({ children, delay = 0 }: { children: React.ReactNode; delay?: number }) {
  const [ref, inView] = useInView({ triggerOnce: true, threshold: 0.2 });
  return (
    <motion.div
      ref={ref}
      initial={{ opacity: 0, y: 50 }}
      animate={inView ? { opacity: 1, y: 0 } : {}}
      transition={{ duration: 0.6, delay }}
    >
      {children}
    </motion.div>
  );
}

// Region card component (with colored placeholder if no image)
function RegionCard({ name, flavor }: { name: string; flavor: string }) {
  // Generate a consistent color based on region name
  const colors = ["bg-amber-700", "bg-green-700", "bg-orange-700", "bg-red-700", "bg-yellow-700", "bg-brown-700"];
  const colorIndex = name.length % colors.length;
  return (
    <Link href={`/regions#${name.toLowerCase()}`} className="block group">
      <div className={`relative h-64 rounded-xl overflow-hidden shadow-lg hover:shadow-2xl transition ${colors[colorIndex]}`}>
        {/* If you have images later, replace this div with <Image> */}
        <div className="absolute inset-0 flex items-center justify-center text-white text-4xl font-bold opacity-30">
          {name.charAt(0)}
        </div>
        <div className="absolute inset-0 bg-gradient-to-t from-black/80 via-black/30 to-transparent" />
        <div className="absolute bottom-0 left-0 right-0 p-5 text-white">
          <h3 className="text-2xl font-bold mb-1">{name}</h3>
          <p className="text-sm opacity-90">{flavor}</p>
          <span className="inline-block mt-2 text-amber-300 font-semibold group-hover:underline">
            Explore Region →
          </span>
        </div>
      </div>
    </Link>
  );
}
